cmake_minimum_required(VERSION 3.0)
project(OHARBaseLayer)
include_directories(/usr/local/include)
set(CMAKE_CXX_FLAGS "-fPIC -g -Wall -std=c++11 ${CMAKE_CXX_FLAGS}")
add_library(${PROJECT_NAME} STATIC ConfigurationDataItem.cpp DataItem.cpp Networker.cpp ProcessorNode.cpp ConfigurationFileReader.cpp Log.cpp NodeConfiguration.cpp DataFileReader.cpp NetworkReader.cpp Package.cpp DataHandler.cpp NetworkWriter.cpp PingHandler.cpp)
set_target_properties(${PROJECT_NAME} PROPERTIES PUBLIC_HEADER "ConfigurationDataItem.h;DataItem.h;NetworkReaderObserver.h;Package.h;ConfigurationFileReader.h;DataReaderObserver.h;NetworkWriter.h;PingHandler.h;DataFileReader.h;Log.h;Networker.h;ProcessorNode.h;DataHandler.h;NetworkReader.h;NodeConfiguration.h")
find_library(BOOSTS boost_system)
target_link_libraries(${PROJECT_NAME} ${BOOSTS})
install(TARGETS ${PROJECT_NAME} ARCHIVE DESTINATION lib LIBRARY DESTINATION lib PUBLIC_HEADER DESTINATION include/${PROJECT_NAME})
